Créer un envoi 
<?php
$ch = curl_init('https://api.colisgo.ca/v1/shipments');

$data = [
    'sender' => [
        'name' => 'ABC Entreprise',
        'address' => '123 Rue Principale, Montréal, QC',
        'phone' => '+15145551234'
    ],
    'receiver' => [
        'name' => 'Jean Tremblay',
        'address' => '456 Rue Saint-Jean, Québec, QC',
        'phone' => '+14185555678'
    ],
    'package' => [
        'weight' => 2.5,
        'type' => 'standard',
        'description' => 'Documents importants'
    ]
];

curl_setopt_array($ch, [
    CURLOPT_RETURNTRANSFER => true,
    CURLOPT_POST => true,
    CURLOPT_POSTFIELDS => json_encode($data),
    CURLOPT_HTTPHEADER => [
        'Authorization: Bearer sk_test_YOUR_SECRET_KEY',
        'Content-Type: application/json'
    ]
]);

$response = curl_exec($ch);
$shipment = json_decode($response, true);
echo "Tracking: " . $shipment['tracking_number'];
